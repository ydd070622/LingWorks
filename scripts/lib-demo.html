function daysDiff(a,b){
  return Math.ceil(
    (new Date(a)-new Date(b))/86400000
  )
}
function fmtDate(d){
  const [y,m,day]=d.split('-')
  return parseInt(m)+'月'+parseInt(day)+'日'
}
